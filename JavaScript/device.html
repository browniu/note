# 获取地理坐标

getPos() {
if (navigator.geolocation) {
  navigator.geolocation.getCurrentPosition(
    position => {
      this.pos = [position.coords.longitude, position.coords.latitude];
    },
    e => {
      throw e.message;
    }
  );
}
}

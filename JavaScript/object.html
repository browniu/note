const deepClone = obj =>{
    let result = Array.isArray(obj)?[]:{}
    for(k in obj){
        if(typeof obj[k] ==== 'object' && obj[k] !== null) result[k] = deepClone(obj[k])
        else result[k] = obj[k]
    }
    return result
}

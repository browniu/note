# 拼接

# 遍历

[1,2,3].forEach(e=>log(e)) // 1 2 3

# 数组转字符串

[1,2,3].join('/') // 1/2/3

# 生成

Array.apply(null,{length:3}) // [undefined,undefined,undefined]

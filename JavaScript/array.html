# 数组

# 拼接

[1,2,3].concat([4,5,6]) //[1,2,3,4,5,6]

# 遍历

[1,2,3].forEach(e=>log(e)) // 1 2 3

# 数组转字符串

[1,2,3].join('/') // 1/2/3

# 生成

Array.apply(null,{length:3}) // [undefined,undefined,undefined]

# 异步遍历

await Promise.all(array.map(async(e)=>console.log(e)))

# 计算

eval([1,2,3].join('+')) //6

# 翻转

[1,2,3].reserve() //[3,2,1]

# 去重

new Set([1,2,3,1,4,1]) //[1,2,3,4]

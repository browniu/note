# TravisCI 持续集成

#.travis.yml

language: node_js
node_js:
  - lts/*

cache: yarn

install:
  - yarn install

script:
  - yarn build

deploy:
  provider: pages
  skip-cleanup: true
  local_dir: docs/.vuepress/dist
  github-token: $GITHUB_TOKEN
  keep-history: true

on:
  branch: master
